+   创建私有组件

    ```
        <div id="#app"></div>

        <template id='my-com'>
            <div>这是我的myCom组件</div>
        </template>


        <script>
            var vm = new Vue({
                el:'#app',
                component:{
                    myCom:{
                        template:'my-com'
                    }
                }
            });
        </script>
    ```
